@pytest.mark.slow
def test_live_single_ticker_smoke():
    """Full pipeline with real LLM + real data APIs. Structure + golden range assertions."""
    graph = create_graph(checkpointer=MemorySaver())
    state = create_initial_state(SMOKE_TICKER)
    result = graph.invoke(
        state,
        {"configurable": {"thread_id": f"smoke-{SMOKE_TICKER}"}},
    )

    # --- Structure assertions (must be present) ---
    assert result.get("judge_recommendation") in ("BUY", "HOLD", "PASS"), \
        f"Unexpected recommendation: {result.get('judge_recommendation')}"
    assert result.get("optimizer_portfolio"), "optimizer_portfolio is empty"
    assert result.get("bull_thesis"), "bull_thesis is empty"
    assert result.get("bear_thesis"), "bear_thesis is empty"
    assert result.get("macro_regime") is not None, "macro_regime is None"

    # --- Judge output with margin_breakdown (D-13) ---
    judge_out = result.get("judge_output")
    assert judge_out is not None, "judge_output is None"
    assert hasattr(judge_out, "margin_breakdown"), "judge_output missing margin_breakdown"
    mb = judge_out.margin_breakdown
    for key in ("oracle", "guardian", "strategist", "base", "raw_total", "clamped"):
        assert key in mb, f"margin_breakdown missing key: {key}"
    assert mb["clamped"] == judge_out.margin_of_safety, \
        f"clamped ({mb['clamped']}) != margin_of_safety ({judge_out.margin_of_safety})"

    # --- Golden range assertions (D-17) ---
    # NOTE (D-18): review if _MARGIN_MIN/_MARGIN_MAX or MAX_POSITION_SIZE change
    conviction = result.get("judge_conviction")
    assert conviction is not None, "judge_conviction is None"
    assert 0.0 <= conviction <= 1.0, f"conviction {conviction} out of [0,1]"

    margin = result.get("judge_margin")
    assert margin is not None, "judge_margin is None"
    assert 0.15 <= margin <= 0.60, f"margin {margin} out of [0.15, 0.60]"

    position_size = result.get("optimizer_metrics", {}).get("position_size", 0.0)
    assert 0.0 <= position_size <= 0.10, f"position_size {position_size} out of [0, 0.10]"

    veto_score = result.get("strategist_veto")
    assert veto_score is not None, "strategist_veto is None"
    assert 0.0 <= veto_score <= 1.0, f"veto_score {veto_score} out of [0,1]"

    # --- Strategist modifier has margin_adjustment (VeTO wired) ---
    strat_mod = result.get("strategist_modifier")
    assert strat_mod is not None, "strategist_modifier is None"
    assert hasattr(strat_mod, "margin_adjustment"), "strategist_modifier missing margin_adjustment"
