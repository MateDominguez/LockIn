  3. Create src/lockin/agents/types.py with the following dataclasses from the Notion Judge spec:

     ```python
     from __future__ import annotations
     from dataclasses import dataclass, field

     @dataclass
     class DataCoverage:
         """Map of Ignorance (Pillar 5) -- tracks what data is available vs missing."""
         available: list[str] = field(default_factory=list)   # e.g. ["10-K 2024", "earnings_call_Q3"]
         missing: list[str] = field(default_factory=list)     # e.g. ["China_regulatory_risk"]
         confidence_impact: float = 0.0                       # e.g. -0.15 = reduces confidence 15%

     @dataclass
     class Signal:
         """A single quantitative signal with optional calibrated base rate."""
         name: str                        # "piotroski_score", "z_score", "vix_level"
         value: float                     # 8.0, 3.2, 35.0
         category: str                    # "high", "safe", "extreme_fear"
         has_base_rate: bool              # True if has calibrated historical frequency
         base_rate: float | None = None   # 0.74 if has_base_rate, else None
         base_rate_source: str | None = None  # "backtest_sp500_2010_2024" or "Piotroski (2000)"

     @dataclass
     class ConfidenceModifier:
         """Output type for Modifier agents (Guardian, Strategist, Oracle).
         They do NOT opine on price -- they opine on how much to trust
         Family 1 (Distribution) agents."""
         margin_adjustment: float         # +0.10 = require 10% more margin of safety
         variance_adjustment: float       # +0.05 = widen distribution 5%
         circuit_breaker: bool            # True = invalid data, hard block (Guardian only)
         circuit_breaker_reason: str | None = None
         signals: list[Signal] = field(default_factory=list)
         data_coverage: DataCoverage = field(default_factory=DataCoverage)
         reasoning: str = ""              # Narrative for glass box

     @dataclass
     class ValueDistribution:
         """Output type for Distribution agents (Value Hunter / Bull, Bear).
         Log-normal distribution parametrized with key percentiles."""
         expected_value: float            # Mean in $ (e.g. $127)
         std_dev: float                   # Standard deviation in $
         p10: float                       # 10th percentile (pessimistic)
         p50: float                       # Median
         p90: float                       # 90th percentile (optimistic)
         confidence: float                # 0.0-1.0, data completeness
         methods_used: list[str] = field(default_factory=list)  # ["EPV", "EVA", "RIM"]
         data_coverage: DataCoverage = field(default_factory=DataCoverage)
         thesis: str = ""                 # Explanatory text (audit trail, NOT calculations)
         key_assumptions: list[str] = field(default_factory=list)

     @dataclass
     class JudgeOutput:
         """Output of the 7-step Judge algorithm."""
         recommendation: str           # BUY | HOLD | PASS
         consensus_distribution: tuple  # (mu, sigma) of LogNormal
         valor_mediano: float
         precio_target: float
         margin_of_safety: float
         p_success: float
         p_base: float
         p_adjustments: dict = field(default_factory=dict)
         kelly_fraction: float = 0.0
         hold_conviction: float = 0.0
         known_unknowns: list[str] = field(default_factory=list)
         convergence_score: float = 0.0
         convergence_alert: bool = False
         bull_weight: float = 0.5
         bear_weight: float = 0.5
         modifiers_applied: dict = field(default_factory=dict)
         circuit_breaker: bool = False
         circuit_breaker_override: bool = False
     ```
