       ```python
       def is_argument_exhausted(prev_thesis: str, curr_thesis: str) -> bool:
           """Detect when Bull-Bear dialectic has stopped producing new arguments.
           Uses Jaccard similarity of top 20 words as proxy (cosine similarity deferred)."""
           if not prev_thesis or not curr_thesis:
               return False
           # Normalize: lowercase, split into words, take top 20 by frequency
           def top_words(text: str, n: int = 20) -> set[str]:
               words = text.lower().split()
               from collections import Counter
               return set(w for w, _ in Counter(words).most_common(n))
           prev_set = top_words(prev_thesis)
           curr_set = top_words(curr_thesis)
           if not prev_set or not curr_set:
               return False
           jaccard = len(prev_set & curr_set) / len(prev_set | curr_set)
           return jaccard > 0.85  # High overlap = arguments exhausted
       ```

     - Wire into graph routing: in the Bull-Bear loop conditional edge, stop if `is_argument_exhausted()` returns True, in addition to `bull_iteration >= MAX_BULL_BEAR_ITERATIONS`:
       ```python
       def should_continue_dialectic(state):
           if state.get("bull_iteration", 0) >= MAX_BULL_BEAR_ITERATIONS:

           prev_thesis = state.get("_prev_bull_thesis", "")
           curr_thesis = state.get("bull_thesis", "")
           if is_argument_exhausted(prev_thesis, curr_thesis):
               return "judge"
           return "bear"  # continue dialectic
       ```
