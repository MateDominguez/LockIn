**1. log_audit_event function** -- writes to audit_logs table via separate psycopg connection (NOT the checkpointer connection -- research confirmed this avoids transaction conflicts).

```python
import json
import psycopg
from datetime import datetime, timezone
from typing import Any

def log_audit_event(
    database_url: str,
    thread_id: str,
    agent_name: str,
    event_type: str,  # "agent_start" | "agent_end"
    state_snapshot: dict,
) -> None:
    """Log an audit event to the audit_logs table.

    Uses a separate short-lived connection to avoid conflicts
    with the LangGraph checkpointer's connection/transactions.
    """
    with psycopg.connect(database_url) as conn:
        with conn.cursor() as cur:
            cur.execute(
                """INSERT INTO audit_logs (thread_id, agent_name, event_type, state_snapshot, timestamp)
                   VALUES (%s, %s, %s, %s, %s)""",
                (
                    thread_id,
                    agent_name,
                    event_type,
                    json.dumps(state_snapshot, default=str),
                    datetime.now(timezone.utc).isoformat(),
                ),
            )
        conn.commit()
```

**2. audit_node wrapper** -- wraps any agent function to add audit logging before and after execution.

```python
from typing import Callable
from langchain_core.runnables import RunnableConfig
from lockin.utils.config import get_settings

def audit_node(agent_name: str, fn: Callable) -> Callable:
    """Wrap an agent function with audit logging."""
    def wrapper(state: dict, config: RunnableConfig) -> dict:
        settings = get_settings()
        thread_id = config.get("configurable", {}).get("thread_id", "unknown")

        # Log start (only log a subset of state to avoid huge payloads)
        log_audit_event(
            settings.database_url,
            thread_id,
            agent_name,
            "agent_start",
            {"asset_ticker": state.get("asset_ticker"), "bull_iteration": state.get("bull_iteration")},
        )

        # Execute agent
        result = fn(state, config)

        # Log end with result
        log_audit_event(
            settings.database_url,
            thread_id,
            agent_name,
            "agent_end",
            result,
        )

        return result

    wrapper.__name__ = f"audit_{agent_name}"
    return wrapper
```

```python
from langgraph.graph import StateGraph, END, START
from langgraph.checkpoint.base import BaseCheckpointSaver
from lockin.graph.state import InvestmentState
from lockin.agents.mock import (
    mock_macro_oracle, mock_value_hunter, mock_bear,
    mock_strategist, mock_guardian, mock_judge, mock_optimizer,
)
from lockin.utils.audit import audit_node

MAX_BULL_BEAR_ITERATIONS = 2

def should_continue_dialectic(state: InvestmentState) -> str:
    """Route after bear: continue to value_hunter if < MAX iterations, else strategist."""
    if state.get("bull_iteration", 0) < MAX_BULL_BEAR_ITERATIONS:
        return "value_hunter"
    return "strategist"

def should_guardian_veto(state: InvestmentState) -> str:
    """Route after guardian: END if veto, else judge."""
    if state.get("guardian_veto", False):
        return "__end__"
    return "judge"

def should_judge_hitl(state: InvestmentState) -> str:
    """Route after judge: END with HITL flag or continue to optimizer."""
    # HITL interrupt is handled INSIDE the judge node (Plan 03).
    # This edge just routes to optimizer after judge completes.
    return "optimizer"

def create_graph(checkpointer: BaseCheckpointSaver | None = None) -> ...:
    """Build and compile the investment analysis StateGraph.

    Graph flow:
    START -> macro_oracle -> value_hunter -> bear -> [conditional: loop or strategist]
    -> strategist -> guardian -> [conditional: veto=END or judge]
    -> judge -> optimizer -> END
    """
    builder = StateGraph(InvestmentState)

    # Add all 7 nodes wrapped with audit logging
    builder.add_node("macro_oracle", audit_node("macro_oracle", mock_macro_oracle))
    builder.add_node("value_hunter", audit_node("value_hunter", mock_value_hunter))
    builder.add_node("bear", audit_node("bear", mock_bear))
    builder.add_node("strategist", audit_node("strategist", mock_strategist))
    builder.add_node("guardian", audit_node("guardian", mock_guardian))
    builder.add_node("judge", audit_node("judge", mock_judge))
    builder.add_node("optimizer", audit_node("optimizer", mock_optimizer))

    # Linear edges
    builder.add_edge(START, "macro_oracle")
    builder.add_edge("macro_oracle", "value_hunter")
    builder.add_edge("value_hunter", "bear")
    builder.add_edge("strategist", "guardian")
    builder.add_edge("optimizer", END)

    # Conditional edges
    builder.add_conditional_edges(
        "bear",
        should_continue_dialectic,
        {"value_hunter": "value_hunter", "strategist": "strategist"},
    )
    builder.add_conditional_edges(
        "guardian",
        should_guardian_veto,
        {"__end__": END, "judge": "judge"},
    )
    # Judge -> optimizer (simple edge for now, HITL added in Plan 03)
    builder.add_edge("judge", "optimizer")

    # Compile
    graph = builder.compile(checkpointer=checkpointer)
    return graph
```
